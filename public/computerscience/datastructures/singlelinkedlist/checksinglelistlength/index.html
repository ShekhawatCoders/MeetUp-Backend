Single Linked List Length(Count Nodes)

Recursive and Iterative Approach

Code : 

//Definining Node class

class Node {

	int data;

	Node next;

	Node(int data) {

		this.data = data;

		this.next = next;

	}

	Node(int data,Node next) {

		this.data = data;

		this.next = next;

	}

}

public class SingleListLength {


	private Node head = null;

	//Insert node at the beginning of the list

	public void insertFirst(int data) {

		Node newNode = new Node(data);

		newNode.next = this.head;

		this.head = newNode;

	}

	//Insert node at the end of the list

	public void insertLast(int data) {

		Node newNode = new Node(data);

		Node temp = this.head;

		while(temp.next != null) {

			temp = temp.next;

		}

		temp.next = newNode;

	}

	//Traverse the whole list

	public void traverse() {

		Node temp = this.head;

		while(temp != null) {

			System.out.print(temp.data+" ");

			temp = temp.next;

		}

		System.out.println();

	}

	//Iterative method to count nodes

	public int checkIterativeLength() {

		int count = 0;

		Node temp = this.head;

		while(temp != null) {

			temp = temp.next;

			count++;

		}

		return count;

	}

	//Calling Recursive method

	public int checkRecursiveLength() {

		return checkRecursiveLengthUtil(this.head);

	}

	//Recursive method to count nodes

	public int checkRecursiveLengthUtil(Node head) {

		if(head == null) return 0;

		else return(1 + checkRecursiveLengthUtil(head.next));

	}

	//Driver code

	public static void main(String[] args) {

		//Instance of this class since we are calling from static method

		SingleListLength sll = new SingleListLength();

		//Calling insert methods

		sll.insertFirst(12);

		sll.insertFirst(7);

		sll.insertLast(182);

		sll.insertFirst(25);

		sll.insertLast(1);

		sll.insertFirst(20);

		sll.traverse();

		//Print counted nodes

		System.out.println(sll.checkRecursiveLength());

		System.out.println(sll.checkIterativeLength());

	}

}
