Heap (PriorityQueue) in Python

Heap in Python

Code : 

//lastModified at 29-August-2020

from heapq import *

list = [10,15,30,20,25,20]

print(list)#before heapify

heapify(list)

print(list)#after heapify

heappush(list,1)#push element in heap

print(list)

print(heappushpop(list,0))#push and then pop

print(list)

print(heapreplace(list,0))#pop and then push

print(list)

print(nlargest(6,list))#6 largest element in heap

print(nsmallest(5,list))#5 smallest element in heap
