import java.io.*;

class Trie {

	Trie[] arr;

	String word;

	boolean leave;

	Trie() {

		arr = new Trie[26];

		word = new String();

		for(int i=0;i<26;i++) {

			arr[i] = null;

		}

		leave = false;

	}

}

public class TrieExample {

	
	Trie root;

	
	TrieExample() {

		root = new Trie();

	}

	
	public void insertWord(String key,String meaning) {

		//assuming key is in lower case letter
 
		int length = key.length();

		
		key = key.toLowerCase();

		
		Trie temp = this.root;

		
		for(int i=0;i < length;i++) {

			
			int index = key.charAt(i) - 'a';

		
			if(temp.arr[index] == null) {

				temp.arr[index] = new Trie();

			}

			
			temp = temp.arr[index];

			
		}

		
		temp.leave = true;

		temp.word = meaning;

		
	}

	
	public String searchWord(String key) {

		
		key = key.toLowerCase();

		
		int length = key.length();

		
		Trie temp = this.root;

		
		for(int i=0;i < length;i++) {

			
			int index = key.charAt(i) - 'a';

			
			if(temp.arr[index] != null) {

				temp = temp.arr[index];

			} else {

				return "Sorry, The Word is not in Dictonary";

			}

			
		}

		
		return temp.word;

		
	}

	

	public static void main(String[] args) throws NumberFormatException, IOException {

		
		TrieExample trie = new TrieExample();

	
		int i = 1;

		
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

		
		while(i != 3) {

			
			System.out.println("Enter 1 to insert data in dictonary , 2 for searching word or 3 to exit
			i = Integer.parseInt(br.readLine());

			String key;

			
			switch(i) {

				case 1:

					key = br.readLine();

					String word = br.readLine();

					trie.insertWord(key, word);

					break;

				case 2:

					key = br.readLine();

					System.out.println(trie.searchWord(key));

					break;

				case 3:

					System.out.println("Thank you !!!");

					break;

					
			}

			
		}
		
	}

}
